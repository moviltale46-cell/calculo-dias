function calcularFecha(){ const valor = document.getElementById( "fechaInicio" ).value; if(!valor){ alert( "Seleccione una fecha." ); return; } // Evita problemas de zona horaria const [anio, mes, dia] = valor.split('-'); let fecha = new Date(anio, mes - 1, dia); // Guardamos la fecha ingresada exactamente const fechaIngresadaTexto = `${dia}/${mes}/${anio}`; // Sumar 45 días fecha.setDate( fecha.getDate()+45 ); // Guardar fecha calculada original const fechaOriginal = new Date( fecha.getFullYear(), fecha.getMonth(), fecha.getDate() ); let motivo = ""; while(true){ const diaSemana = fecha.getDay(); if(diaSemana===6){ motivo = "sábado"; fecha.setDate( fecha.getDate()+1 ); continue; } if(diaSemana===0){ motivo = "domingo"; fecha.setDate( fecha.getDate()+1 ); continue; } if(esFeriado(fecha)){ motivo = "feriado"; fecha.setDate( fecha.getDate()+1 ); continue; } break; } let mensaje=""; if( fechaOriginal.getTime() === fecha.getTime() ){ mensaje = "✅ La fecha calculada es un día hábil."; }else{ mensaje = `⚠️ La fecha calculada cayó en ${motivo}. Se trasladó al siguiente día hábil.`; } document.getElementById( "resultado" ).innerHTML = `

Fecha ingresada: ${fechaIngresadaTexto}

Fecha +45 días: ${formatear(fechaOriginal)}

Fecha hábil final: ${formatear(fecha)}

Estado: ${mensaje}

`; }
